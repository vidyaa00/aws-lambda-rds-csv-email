import os
import pg8000
import csv
import io
import boto3

def lambda_handler(event, context):
    
    try:
        print("Lambda triggered")
        # DB connection
        conn = pg8000.connect(
            host=os.environ['DB_HOST'],
            user=os.environ['DB_USER'],
            password=os.environ['DB_PASSWORD'],
            database=os.environ['DB_NAME'],
            port=5432,
            ssl_context=True
        )

        cursor = conn.cursor()
        cursor.execute("""
            SELECT 
                p.PTY_FirstName,
                p.PTY_LastName,
                p.PTY_Phone,
                a.Add_City,
                s.Stt_Name
            FROM OPT_Party p
            JOIN OPT_Address a ON p.PTY_ID = a.Add_PartyID
            JOIN SYS_State s ON a.Add_State = s.Stt_ID
        """)
        rows = cursor.fetchall()

        # Create CSV
        output = io.StringIO()
        writer = csv.writer(output)
        writer.writerow(['First Name', 'Last Name', 'Phone', 'City', 'State'])
        writer.writerows(rows)
        csv_data = output.getvalue()

        cursor.close()
        conn.close()

        # Send email via SES
        ses = boto3.client('ses')


        sender = os.environ['SENDER_EMAIL']
        receiver1 = os.environ['RECEIVER_EMAIL_1']
        receiver2 = os.environ['RECEIVER_EMAIL_2']

        email_body = f"""From: {sender}
Cc: {receiver1}
To:{receiver2}
Subject: Client Report â€“ Generated by AWS Lambda
MIME-Version: 1.0
Content-Type: multipart/mixed; boundary="NextPart"

--NextPart
Content-Type: text/plain

Hello,

This email was automatically generated by an AWS Lambda function 
as part of a cloud-based automation system.

The attached CSV file contains client information securely retrieved 
from the database.

Regards,
AWS Cloud Automation System

--NextPart
Content-Type: text/csv;charset="UTF-8"
Content-Disposition: attachment; filename="clients.csv"

{csv_data}
--NextPart--
"""

        ses.send_raw_email(
            Source=sender,
            Destinations=[receiver1,receiver2],
            RawMessage={'Data': email_body}
        )
        print("Email sent successfully")

        return {
            "statusCode": 200,
            "headers": {
                "Content-Type": "text/plain"
            },
            "body": "Request processed successfully. Email has been sent."
        }


    except Exception as e:
        return {
            "statusCode": 500,
            "error": str(e)
        }
